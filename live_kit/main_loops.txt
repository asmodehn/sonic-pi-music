#  ---- ----  ----  ---- live loops synchro
use_bpm 120

set :beats, 2
set :sub_beat, 4


set :chords, [
  (chord :C, :major),
  (chord :A, :minor),
  (chord :D, :minor),
  (chord :G, :dom7)
].ring

set :chrd, (chord :C, :major)


# ---- live loops

live_loop :bar_beats do
  set :chrd, (get :chords).tick
  cue :bar
  (get :beats).times do
    cue :beat
    sleep 1
  end
end
